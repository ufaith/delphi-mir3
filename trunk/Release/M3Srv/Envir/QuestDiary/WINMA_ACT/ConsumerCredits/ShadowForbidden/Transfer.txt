[@main]
#ACT
;;SET [920] 0
MOV A0 %USERNAME
FormatStr "FLD_CHARNAME='%s'" %USERNAME
ReadValueSql "TBL_CHARACTER" %A9 "FLD_ANYINC" @N_6_AN

[@N_6_AN()]
#ACT
MOV 暗影层数 %ARG(1)
LOADVALUE D8 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 总层数
#IF
!EQUAL 暗影层数 %D8
#ACT
MOV D8 %暗影层数
INC 暗影层数 1
FormatStr "FLD_CHARNAME='%s'" %USERNAME
mov A5 %A9
FormatStr "FLD_ANYINC='%s'" %暗影层数
UPDATEVALUESQL "TBL_CHARACTER" %A5 %A9
MOV A7 "第"
ADDSTR A7 %暗影层数
ADDSTR A7 "层积分"
loadvalue D0 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control %A7
loadvalue  D7 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 选择时间
goto @进入计时_ANX
#ELSEACT
goto @得奖_ANX

[@进入计时_ANX]
#act
mov A0 %USERNAME 
MOV A7 "第"
ADDSTR A7 %暗影层数
ADDSTR A7 "层积分"
MOV A6 "第"
ADDSTR A6 %暗影层数
ADDSTR A6 "层地图"
loadvalue D0 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control %A7
loadvalue A1 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control %A6
#IF
CHECKJFPOINT %D0
#ACT
;;SET [920] 1
Delaygoto [local] 5 @计时开始_ANc
loadvalue  D7 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 进入时间
#SAY
恭喜<$OUTPUT(A0)>,你现在是闯过<$OUTPUT(D8)>关!\
进入<$OUTPUT(暗影层数)>层.需要<$OUTPUT(D0)>点积分!\
你有{FCOLOR=250}<$OUTPUT(D7)>{FCOLOR=255}秒在此逗留,完了就到下一层了!\
{FCOLOR=249}快点去收拾家当哦!{FCOLOR=255}\
#ELSEACT
SYSMSG "对不起,你得积分不足!"
goto @计时开始_ANc

[@计时开始_ANc]
#ACT
loadvalue  D7 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 进入时间
mov 计时器 %D7
close
GOTO @测测时间_ANX

[@测测时间_ANX]
#IF
SMALL 计时器 1
#ACT
GOTO @进入_ANX
#ELSEACT
GOTO @倒数时间_ANX

[@倒数时间_ANX]
#ACT
SYSMSG "还剩下<$OUTPUT(计时器)>秒!"
DEC 计时器 1
Delaygoto [local] 2 @测测时间_ANX

[@出去_ANX]
#ACT
loadvalue A1 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 回去地图代码  
loadvalue D2 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control X坐标  
loadvalue D3 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control Y坐标 
MAPMOVE %A1 %D2 %D3
SYSMSG "回去顺风哦!"
Breaktimerecall
break

[@进入_ANX]
#act
mov A0 %USERNAME 
MOV A7 "第"
ADDSTR A7 %暗影层数
ADDSTR A7 "层积分"
MOV A6 "第"
ADDSTR A6 %暗影层数
ADDSTR A6 "层地图"
loadvalue D0 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control %A7
loadvalue A1 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control %A6
#if
CHECKJFPOINT %D0
#act
DECJFPOINT %D0
mapmove  %A1
SysMsg  "你花费了<$OUTPUT(D0)>点积分进入夺宝禁区<$OUTPUT(暗影层数)>层,祝您好运.您还剩余<$NEWJFPOINT>点积分！
#ELSEACT
SYSMSG  "对不起,你的积分不足!"
GOTO @出去_ANX

[@计时开始_ANX]
#ACT
loadvalue D7 "QuestDiary\WINMA_DATA\Control\ShadowForbidden.txt" Control 进入时间
mov 计时器 %D7
close
GOTO @测测时间_ANc

[@测测时间_ANc]
#IF
SMALL 计时器 1
#ACT
GOTO @出去_ANX
#ELSEACT
SYSMSG "恭喜你过关,你还剩下<$OUTPUT(计时器)>秒在这里停留!"
DEC 计时器 1
Delaygoto [local] 2 @测测时间_ANc

[@得奖_ANX]
#IF
CHECK [920] 0
#SAY
{FCOLOR=250}<$USERNAME>{FCOLOR=255} 恭喜你挑战成功\\
系统将奖励你 {FCOLOR=250}2块{FCOLOR=255} 封魔石和 {FCOLOR=250}20W金币{FCOLOR=255}\
以及{FCOLOR=250}1个{FCOLOR=255} 地王令 \\
感谢您的参与,同时您的 {FCOLOR=250}声望{FCOLOR=255} 提升4点\\
#ACT
SET [920] 1
ADDSHENGWANG 4
give 封魔石 2
;give 神秘钥匙 1
give 金币 200000
give 地王令 1
lineMsg [Grobal] 一位伟大的勇士'<$USERNAME>'成功通过[暗影禁地]重重考验
sysmsg "恭喜您 <$USERNAME>,获得了2块封魔石+20W金币,声望提升4点,地王令一块"
Delaygoto [local] 3 @计时开始_ANX
break
#ELSEACT
SYSMSG "发现[双杀数据]系统终止"
GOTO @出去_ANX
BREAK
;;EI3G-网游先锋传奇3商业版本-纯1.45EI版
